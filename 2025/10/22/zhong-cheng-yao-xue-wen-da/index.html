    // 打乱数组顺序（随机出题核心逻辑）
    function shuffleArray(array) {
        const newArray = [...array];
        for (let i = newArray.length - 1; i > 0; i--) {
            const j = Math.floor(Math.random() * (i + 1));
            [newArray[i], newArray[j]] = [newArray[j], newArray[i]];
        }
        return newArray;
    }

    // 初始化变量
    let shuffledDrugs = [];
    let currentIndex = 0;

    // DOM元素
    const drugNameEl = document.querySelector('.drug-name');
    const answerEl = document.querySelector('.answer');
    const showAnswerBtn = document.getElementById('show-answer');
    const nextQuestionBtn = document.getElementById('next-question');

    // 初始化游戏
    function initGame() {
        shuffledDrugs = shuffleArray(drugsData);
        currentIndex = 0;
    }

    // 显示下一题（随机）
    function showNextQuestion() {
        if (currentIndex < shuffledDrugs.length) {
            const currentDrug = shuffledDrugs[currentIndex];
            drugNameEl.textContent = currentDrug.name;
            answerEl.style.display = 'none';
            currentIndex++;
        } else {
            // 所有题目完成后重新洗牌
            drugNameEl.textContent = '已完成所有题目，将重新开始！';
            answerEl.style.display = 'none';
            setTimeout(initGame, 1000);
            setTimeout(showNextQuestion, 1000);
        }
    }

    // 显示答案（包含章节信息）
    function showAnswer() {
        if (currentIndex > 0 && currentIndex <= shuffledDrugs.length) {
            const currentDrug = shuffledDrugs[currentIndex - 1];
            answerEl.innerHTML = `
                <strong>所属小类：</strong>${currentDrug.category}<br>
                <strong>所属章节：</strong>${currentDrug.chapter}
            `;
            answerEl.style.display = 'block';
        }
    }

    // 绑定事件
    nextQuestionBtn.addEventListener('click', showNextQuestion);
    showAnswerBtn.addEventListener('click', showAnswer);

    // 初始化游戏
    initGame();
</script>
